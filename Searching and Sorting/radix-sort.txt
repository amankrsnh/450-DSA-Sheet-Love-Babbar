void countSort(int arr[],int n,int pos)
{
    int output[n]={0},count[10]={0};
    for(int i=0;i<n;++i)
        count[(arr[i]/pos)%10]+=1;
    for(int i=1;i<10;++i)
        count[i]+=count[i-1];
    for(int i=n-1;i>=0;--i)
    {
        output[count[(arr[i]/pos)%10]-1]=arr[i];
        --count[(arr[i]/pos)%10];
    }
    for(int i=0;i<n;++i)
        arr[i]=output[i];
}
void radixSort(int arr[], int n) 
{ 
   int max=*max_element(arr,arr+n);
   for(int pos=1;(max/pos)>0;pos*=10)
        countSort(arr,n,pos);
}

int main() 
{ 
    int t;

    {

        int arr[n];
        
        for(int i=0;i<n;i++)
            cin>>arr[i];
  
        radixSort(arr, n);
